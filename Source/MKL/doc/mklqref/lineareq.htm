Solving Systems of Linear Equations

?getrs solves a system of linear equations with an LU-factored square matrix, with multiple right-hand sides.
call sgetrs (trans, n, nrhs, SA, lda, IPIV, SB, ldb, info)
call dgetrs (trans, n, nrhs, DA, lda, IPIV, DB, ldb, info)
call cgetrs (trans, n, nrhs, CA, lda, IPIV, CB, ldb, info)
call zgetrs (trans, n, nrhs, ZA, lda, IPIV, ZB, ldb, info)

?gbtrs solves a system of linear equations with an LU-factored band matrix, with multiple right-hand sides.
call sgbtrs (trans, n, kl, ku, nrhs, SAB, ldab, IPIV, SB, ldb, info)
call dgbtrs (trans, n, kl, ku, nrhs, DAB, ldab, IPIV, DB, ldb, info)
call cgbtrs (trans, n, kl, ku, nrhs, CAB, ldab, IPIV, CB, ldb, info)
call zgbtrs (trans, n, kl, ku, nrhs, ZAB, ldab, IPIV, ZB, ldb, info)

?potrs solves a system of linear equations with a Cholesky-factored symmetric (Hermitian) positive-definite matrix, with multiple right-hand sides.
call spotrs (uplo, n, nrhs, SA, lda, SB, ldb, info)
call dpotrs (uplo, n, nrhs, DA, lda, DB, ldb, info)
call cpotrs (uplo, n, nrhs, CA, lda, CB, ldb, info)
call zpotrs (uplo, n, nrhs, ZA, lda, ZB, ldb, info)
