Matrix Inversion

?getri computes the inverse of a general matrix that has been factored by ?getrf.
call sgetri (n, SA, lda, IPIV, SWORK, lwork, info)
call dgetri (n, DA, lda, IPIV, DWORK, lwork, info)
call cgetri (n, CA, lda, IPIV, CWORK, lwork, info)
call zgetri (n, ZA, lda, IPIV, ZWORK, lwork, info)

?potri computes the inverse of a symmetric (Hermitian) positive-definite matrix that has been factored by ?potrf.
call spotri (uplo, n, SA, lda, info)
call dpotri (uplo, n, DA, lda, info)
call cpotri (uplo, n, CA, lda, info)
call zpotri (uplo, n, ZA, lda, info)

?pptri computes the inverse of a packed symmetric (Hermitian) positive-definite matrix that has been factored by ?pptrf.
call spptri (uplo, n, SAP, info)
call dpptri (uplo, n, DAP, info)
call cpptri (uplo, n, CAP, info)
call zpptri (uplo, n, ZAP, info)

?sytri computes the inverse of a symmetric matrix that has been factored by ?sytrf.
call ssytri (uplo, n, SA, lda, IPIV, SWORK, info)
call dsytri (uplo, n, DA, lda, IPIV, DWORK, info)
call csytri (uplo, n, CA, lda, IPIV, CWORK, info)
call zsytri (uplo, n, ZA, lda, IPIV, ZWORK, info)

?hetri computes the inverse of a complex Hermitian matrix that has been factored by ?hetrf.
call chetri (uplo, n, CA, lda, IPIV, CWORK, info)
call zhetri (uplo, n, ZA, lda, IPIV, ZWORK, info)

?sptri computes the inverse of a symmetric matrix using packed storage that has been factored by ?sptrf.
call ssptri (uplo, n, SAP, IPIV, SWORK, info)
call dsptri (uplo, n, DAP, IPIV, DWORK, info)
call csptri (uplo, n, CAP, IPIV, CWORK, info)
call zsptri (uplo, n, ZAP, IPIV, ZWORK, info)

?hptri computes the inverse of a complex Hermitian matrix using packed storage that has been factored by ?hptrf.
call chptri (uplo, n, CAP, IPIV, CWORK, info)
call zhptri (uplo, n, ZAP, IPIV, ZWORK, info)

?trtri computes the inverse of a triangular matrix.
call strtri (uplo, diag, n, SA, lda, info)
call dtrtri (uplo, diag, n, DA, lda, info)
call ctrtri (uplo, diag, n, CA, lda, info)
call ztrtri (uplo, diag, n, ZA, lda, info)

?tptri computes the inverse of a triangular matrix using packed storage.
call stptri (uplo, diag, n, SAP, info)
call dtptri (uplo, diag, n, DAP, info)
call ctptri (uplo, diag, n, CAP, info)
call ztptri (uplo, diag, n, ZAP, info)
