QR and LQ Factorizations

?geqrf computes the QR factorization of a general m by n matrix.
call sgeqrf ( m, n, SA, lda, STAU, SWORK, lwork, info )
call dgeqrf ( m, n, DA, lda, DTAU, DWORK, lwork, info )
call cgeqrf ( m, n, CA, lda, CTAU, CWORK, lwork, info )
call zgeqrf ( m, n, ZA, lda, ZTAU, ZWORK, lwork, info )

?geqpf computes the QR factorization of a general m by n matrix with pivoting.
call sgeqpf ( m, n, SA, lda, JPVT, STAU, SWORK, info )
call dgeqpf ( m, n, DA, lda, JPVT, DTAU, DWORK, info )
call cgeqpf ( m, n, CA, lda, JPVT, CTAU, CWORK, SRWORK, info )
call zgeqpf ( m, n, ZA, lda, JPVT, ZTAU, ZWORK, DRWORK, info )

?orgqr generates the real orthogonal matrix Q of the QR factorization formed by ?geqrf or ?geqpf.
call sorgqr (m, n, k, SA, lda, STAU, SWORK, lwork, info)
call dorgqr (m, n, k, DA, lda, DTAU, DWORK, lwork, info)

?ormqr multiplies a real matrix by the orthogonal matrix Q of the QR factorization formed by ?geqrf or ?geqpf.
call sormqr (side, trans, m, n, k, SA, lda, STAU, SC, ldc, SWORK, lwork, info)
call dormqr (side, trans, m, n, k, DA, lda, DTAU, DC, ldc, DWORK, lwork, info)

?ungqr generates the complex unitary matrix Q of the QR factorization formed by ?geqrf or ?geqpf.
call cungqr (m, n, k, CA, lda, CTAU, CWORK, lwork, info)
call zungqr (m, n, k, ZA, lda, ZTAU, ZWORK, lwork, info)

?unmqr multiplies a complex matrix by the unitary matrix Q of the QR factorization formed by ?geqrf or ?geqpf.
call cunmqr (side, trans, m, n, k, CA, lda, CTAU, CC, ldc, CWORK, lwork, info)
call zunmqr (side, trans, m, n, k, ZA, lda, ZTAU, ZC, ldc, ZWORK, lwork, info)

?gelqf computes the LQ factorization of a general m by n matrix.
call sgelqf (m, n, SA, lda, STAU, SWORK, lwork, info)
call dgelqf (m, n, DA, lda, DTAU, DWORK, lwork, info)
call cgelqf (m, n, CA, lda, CTAU, CWORK, lwork, info)
call zgelqf (m, n, ZA, lda, ZTAU, ZWORK, lwork, info)

?orglq generates the real orthogonal matrix Q of the LQ factorization formed by ?gelqf.
call sorglq (m, n, k, SA, lda, STAU, SWORK, lwork, info)
call dorglq (m, n, k, DA, lda, DTAU, DWORK, lwork, info)

?ormlq multiplies a real matrix by the orthogonal matrix Q of the LQ factorization formed by ?gelqf.
call sormlq (side, trans, m, n, k, SA, lda, STAU, SC, ldc, SWORK, lwork, info)
call dormlq (side, trans, m, n, k, DA, lda, DTAU, DC, ldc, DWORK, lwork, info)

?unglq generates the complex unitary matrix Q of the LQ factorization formed by ?gelqf.
call cunglq (m, n, k, CA, lda, CTAU, CWORK, lwork, info)
call zunglq (m, n, k, ZA, lda, ZTAU, ZWORK, lwork, info)

?unmlq multiplies a complex matrix by the unitary matrix Q of the LQ factorization formed by ?gelqf.
call cunmlq (side, trans, m, n, k, CA, lda, CTAU, CC, ldc, CWORK, lwork, info)
call zunmlq (side, trans, m, n, k, ZA, lda, ZTAU, ZC, ldc, ZWORK, lwork, info)
