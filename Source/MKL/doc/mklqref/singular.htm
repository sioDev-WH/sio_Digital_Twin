Singular Value Decomposition

?gebrd reduces a general matrix to bidiagonal form by an orthogonal (unitary) transformation.
call sgebrd (m, n, SA, lda, SD, SE, STAUQ, STAUP, WORK, lwork, info)
call dgebrd (m, n, DA, lda, DD, DE, DTAUQ, DTAUP, WORK, lwork, info)
call cgebrd (m, n, CA, lda, SD, SE, CTAUQ, CTAUP, WORK, lwork, info)
call zgebrd (m, n, ZA, lda, DD, DE, ZTAUQ, ZTAUP, WORK, lwork, info)

?gbbrd reduces a general band matrix to bidiagonal form by an orthogonal (unitary) transformation.
call sgbbrd (vect, m, n, ncc, kl, ku, SAB, ldab, SD, SE, SQ, ldq, SPT, ldpt, SC, ldc, SWORK, info)
call dgbbrd (vect, m, n, ncc, kl, ku, DAB, ldab, DD, DE, DQ, ldq, DPT, ldpt, DC, ldc, DWORK, info)
call cgbbrd (vect, m, n, ncc, kl, ku, CAB, ldab, SD, SE, CQ, ldq, CPT, ldpt, CC, ldc, CWORK, SRWORK, info)
call zgbbrd (vect, m, n, ncc, kl, ku, ZAB, ldab, DD, DE, ZQ, ldq, ZPT, ldpt, ZC, ldc, ZWORK, DRWORK, info)

?orgbr generates the real orthogonal matrix Q or PT determined by ?gebrd.
call sorgbr (vect, m, n, k, SA, lda, STAU, SWORK, lwork, info)
call dorgbr (vect, m, n, k, DA, lda, DTAU, DWORK, lwork, info)

?ormbr multiplies an arbitrary real matrix by the real orthogonal matrix Q or P determined by ?gebrd.
call sormbr (vect, side, trans, m, n, k, SA, lda, STAU, SC, ldc, SWORK, lwork, info)
call dormbr (vect, side, trans, m, n, k, DA, lda, DTAU, DC, ldc, DWORK, lwork, info)

?ungbr generates the complex unitary matrix Q or PH determined by ?gebrd.
call cungbr (vect, m, n, k, CA, lda, CTAU, CWORK, lwork, info)
call zungbr (vect, m, n, k, ZA, lda, ZTAU, ZWORK, lwork, info)

?unmbr multiplies an arbitrary complex matrix by the unitary matrix Q or P determined by ?gebrd.
call cunmbr (vect, side, trans, m, n, k, CA, lda, CTAU, CC, ldc, CWORK, lwork, info)
call zunmbr (vect, side, trans, m, n, k, ZA, lda, ZTAU, ZC, ldc, ZWORK, lwork, info)

?bdsqr computes the singular value decomposition of a general matrix that has been reduced to bidiagonal form.
call sbdsqr (uplo, n, ncvt, nru, ncc, SD, SE, SVT, ldvt, SU, ldu, SC, ldc, SWORK, info)
call dbdsqr (uplo, n, ncvt, nru, ncc, DD, DE, DVT, ldvt, DU, ldu, DC, ldc, DWORK, info)
call cbdsqr (uplo, n, ncvt, nru, ncc, SD, SE, CVT, ldvt, CU, ldu, CC, ldc, CWORK, info)
call zbdsqr (uplo, n, ncvt, nru, ncc, DD, DE, ZVT, ldvt, ZU, ldu, ZC, ldc, ZWORK, info)
