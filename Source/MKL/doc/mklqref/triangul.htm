Triangular Factorization

?getrf computes the LU factorization of a general matrix.
call sgetrf (m, n, SA, lda, IPIV, info)
call dgetrf (m, n, DA, lda, IPIV, info)
call cgetrf (m, n, CA, lda, IPIV, info)
call zgetrf (m, n, ZA, lda, IPIV, info)

?gbtrf computes the LU factorization of a general band matrix.
call sgbtrf (m, n, kl, ku, SAB, ldab, IPIV, info)
call dgbtrf (m, n, kl, ku, DAB, ldab, IPIV, info)
call cgbtrf (m, n, kl, ku, CAB, ldab, IPIV, info)
call zgbtrf (m, n, kl, ku, ZAB, ldab, IPIV, info)

?potrf computes the Cholesky factorization of a symmetric (Hermitian) positive-definite matrix.
call spotrf (uplo, n, SA, lda, info)
call dpotrf (uplo, n, DA, lda, info)
call cpotrf (uplo, n, CA, lda, info)
call zpotrf (uplo, n, ZA, lda, info)

?pptrf computes the Cholesky factorization of a symmetric (Hermitian) positive-definite matrix using packed storage.
call spptrf (uplo, n, SAP, info)
call dpptrf (uplo, n, DAP, info)
call cpptrf (uplo, n, CAP, info)
call zpptrf (uplo, n, ZAP, info)

?pbtrf computes the Cholesky factorization of a symmetric (Hermitian) positive-definite band matrix.
call spbtrf (uplo, n, kd, SAB, ldab, info)
call dpbtrf (uplo, n, kd, DAB, ldab, info)
call cpbtrf (uplo, n, kd, CAB, ldab, info)
call zpbtrf (uplo, n, kd, ZAB, ldab, info)

?sytrf computes the Bunch-Kaufman factorization of a symmetric matrix.
call ssytrf (uplo, n, SA, lda, IPIV, SWORK, lwork, info)
call dsytrf (uplo, n, DA, lda, IPIV, DWORK, lwork, info)
call csytrf (uplo, n, CA, lda, IPIV, CWORK, lwork, info)
call zsytrf (uplo, n, ZA, lda, IPIV, ZWORK, lwork, info)

?hetrf computes the Bunch-Kaufman factorization of a complex Hermitian matrix.
call chetrf (uplo, n, CA, lda, IPIV, CWORK, lwork, info)
call zhetrf (uplo, n, ZA, lda, IPIV, ZWORK, lwork, info)

?sptrf computes the Bunch-Kaufman factorization of a symmetric matrix using packed storage.
call ssptrf (uplo, n, SAP, IPIV, info)
call dsptrf (uplo, n, DAP, IPIV, info)
call csptrf (uplo, n, CAP, IPIV, info)
call zsptrf (uplo, n, ZAP, IPIV, info)

?hptrf computes the Bunch-Kaufman factorization of a complex Hermitian matrix using packed storage.
call chptrf (uplo, n, CAP, IPIV, info)
call zhptrf (uplo, n, ZAP, IPIV, info)
